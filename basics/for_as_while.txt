package basics

import (
	"fmt"
	"time"
)

func main() {
	start := time.Now()
	i := 1
	for i <= 5 {
		fmt.Println(i)
		i++
	}
	sum := 0
	for {
		sum += 10
		fmt.Println("Sum:", sum)
		if sum >= 50 {
			break
		}
	}
	num := 1
	for num <= 10 {
		if num%2 == 0 {
			num++
			continue
		}
		fmt.Println(num)
		num++
	}

	fmt.Printf("Execution Time: %s\n", time.Since(start))
}
